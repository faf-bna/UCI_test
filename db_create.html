Informacion:

Datos: Utilización de la capacidad instalada en la industria, nivel general y bloques sectoriales.

Notas: - El bloque de Productos alimenticios y bebidas se calcula sin considerar la capacidad instalada en la actividad vitivinícola ni en los ingenios azucareros. El bloque de Sustancias y productos químicos se calcula sin considerar la utilización de la capacidad instalada de la industria farmaceútica.
- Los promedios pueden no coincidir por redondeo en las cifras parciales.

Fuente: INDEC.

Libraries

library(xlsx)
library(tidyverse)
library(lubridate)

Download and import data from INDEC

# Metodologias
download.file("https://www.indec.gob.ar/ftp/cuadros/economia/notas_metodologicas_capacidad_instalada_base2004.pdf",
              "./met/notas_metodologicas_capacidad_instalada_base2004.pdf")

download.file("https://www.indec.gob.ar/ftp/cuadros/economia/sintesis_metodologica_capacidad_instalada.pdf",
              "./met/sintesis_metodologica_capacidad_instalada.pdf")

# Download ----
download.file("https://www.indec.gob.ar/ftp/cuadros/economia/sh_capacidad_08_20.xls",
              "./data/sh_capacidad_08_20.xlsx")

# Import ----
uci <- read.xlsx("./data/sh_capacidad_08_20.xlsx", 
          sheetIndex = 1, startRow = 2, header = TRUE)

# Clean ----
uci <- uci %>%
  as_tibble() %>%
  select(1:14) %>%
  filter_all(any_vars(!is.na(.))) %>%
  filter(!(str_detect(Período, "Año"))) %>%
  slice(1:(n()-4)) %>%
  bind_cols(periodo = seq(ymd("2016-1-1"), ymd(Sys.Date()), by = "months") %>%
  .[1:(length(.)-3)]
  ) %>%
  select(periodo, everything(), -Período) 

# Sectores ----
sectores <- uci %>%
  colnames() %>%
  .[-1] %>%
  as_tibble() %>%
  bind_cols(cod = LETTERS[1:nrow(.)]) %>%
  select(cod, value) %>%
  rename(sector = value) %>%
  mutate(sector = str_replace_all(sector, regex("\\W+"), " "))

# UCI database con cods:
uci <- uci %>%
  setNames(c(colnames(.)[1], sectores$cod)) %>%
  gather(key = "sector", value = "uci", -periodo) %>%
  mutate(uci = as.numeric(uci))
## Warning: attributes are not identical across measure variables;
## they will be dropped
# Save data as RDS
saveRDS(uci, "./data/uci.RDS")
saveRDS(sectores, "./data/sectores.RDS")
